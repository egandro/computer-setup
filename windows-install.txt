rem ssh
rem https://stackoverflow.com/questions/16212816/setting-up-openssh-for-windows-using-public-key-authentication
---
# 1. Install OpenSSH Server
Add-WindowsCapability -Online -Name OpenSSH.Server~~~~0.0.1.0

# 2. Set Services to start automatically (do this early)
Set-Service -Name ssh-agent -StartupType Automatic
Set-Service -Name sshd -StartupType Automatic

# 3. Ensure the service is STOPPED before editing config to avoid file locks
Stop-Service sshd -ErrorAction SilentlyContinue

# 4. Define Config Path
$ConfigPath = "$Env:PROGRAMDATA\ssh\sshd_config"

# 5. Backup the default config (Good practice)
if (-not (Test-Path "$ConfigPath.bak")) {
    Copy-Item $ConfigPath "$ConfigPath.bak"
}

# 6. Apply Configuration Changes
# We read the content, perform replacements, and save it back.
$Content = Get-Content $ConfigPath

# Comment out the default administrative key block so Admins can use regular .ssh/authorized_keys
# We replace the specific block header to comment it out.
$Content = $Content -replace 'Match Group administrators', '#Match Group administrators'

# Fix the AuthorizedKeysFile lookup to ensure it looks at the user's home directory
$Content = $Content -replace 'AuthorizedKeysFile __PROGRAMDATA__', '#AuthorizedKeysFile __PROGRAMDATA__'

# Save the modifications
$Content | Set-Content $ConfigPath

# 7. Append your specific security settings
# We use logic to ensure we don't append the same lines multiple times if you run the script twice.
$Settings = @(
    "AuthorizedKeysFile .ssh/authorized_keys",
    "PasswordAuthentication no",
    "PubkeyAuthentication yes"
)

foreach ($Line in $Settings) {
    # Only add the line if it doesn't already exist in the file
    $Pattern = [Regex]::Escape($Line)
    if (-not ($Content -match $Pattern)) {
        Add-Content $ConfigPath $Line
    }
}

# 8. Set Default Shell to PowerShell (Optional but highly recommended)
# Without this, SSH defaults to the old CMD Command Prompt.
New-ItemProperty -Path "HKLM:\SOFTWARE\OpenSSH" -Name DefaultShell -Value "C:\Windows\System32\WindowsPowerShell\v1.0\powershell.exe" -PropertyType String -Force

# 9. Verify Firewall Rule (Usually automatic, but good to be safe)
if (-not (Get-NetFirewallRule -Name "OpenSSH-Server-In-TCP" -ErrorAction SilentlyContinue)) {
    New-NetFirewallRule -Name 'OpenSSH-Server-In-TCP' -DisplayName 'OpenSSH Server (sshd)' -Enabled True -Direction Inbound -Protocol TCP -Action Allow -LocalPort 22
}

# 10. Start the services (Crucial: This loads the new config)
Start-Service ssh-agent
Start-Service sshd
---
